      'Header':Header
    },
    methods: {
        call_num(){
            this.is_show = true

        fulfill(id){
            let formdata = new FormData()
            formdata.append('id',id)
            this.axios({
                url:'http://127.0.0.1:8000/api/delete/',
                method:'post',
                data:formdata
            }).then(res=>{
                if (res.data.code == 200){
                    window.location.reload() 
                }
            })
        },
        down(id){
            let formdata = new FormData()
            formdata.append('id',id)
            this.axios({
                url:'http://127.0.0.1:8000/api/down_num/',
                method:'post',
                data:formdata
            }).then(res=>{
                if (res.data.code == 200){
                    this.one_user = res.data.data
                }
            })
        }

    },
    mounted() {
        this.axios({
            url:'http://127.0.0.1:8000/api/call_num/',
            method:'get'
        }).then(res=>{
            if (res.data.code == 200){
                this.one_user = res.data.data
            }

        })
    }
}
